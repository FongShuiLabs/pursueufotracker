name: Poll war.gov for new PURSUE drops

# Polls the official war.gov UAP CSV every 30 minutes during US weekday business
# hours and hourly otherwise. Opens an issue when the CSV actually changes
# (CSV hash diff), or when CSV has been unreachable for 12 consecutive polls
# (~6 hours of cron coverage). HTML envelope hash is NOT used as a change
# signal because the page's CSRF tokens churn every fetch.

on:
  schedule:
    # Every 30 min, weekdays 13:00-22:00 UTC (9am-6pm ET)
    - cron: '*/30 13-22 * * 1-5'
    # Hourly off-hours, 7 days a week
    - cron: '0 0-12,23 * * *'
  workflow_dispatch:

permissions:
  contents: write
  issues: write

concurrency:
  group: poll-wargov
  cancel-in-progress: false

jobs:
  poll:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout
        uses: actions/checkout@v4
        with:
          fetch-depth: 1

      - name: Set up Python
        uses: actions/setup-python@v5
        with:
          python-version: '3.12'
          cache: 'pip'

      - name: Install poller deps
        run: |
          pip install curl_cffi playwright
          # Install Chromium only (not all browsers) to keep the install fast
          # and small. With-deps ensures Akamai-targeted libs (fonts, etc.).
          python -m playwright install --with-deps chromium

      - name: Poll war.gov
        id: poll
        continue-on-error: true
        run: python -m pipeline.poll_wargov

      - name: Commit poll state
        if: steps.poll.outcome == 'success' || steps.poll.outputs.new_drop == 'true' || steps.poll.outputs.csv_stuck == 'true'
        run: |
          git config user.name "pursueufotracker-bot"
          git config user.email "bot@pursueufotracker.com"
          git add data/poll-state.json
          if git diff --cached --quiet; then
            echo "No state change to commit."
          else
            if [ "${{ steps.poll.outputs.new_drop }}" = "true" ]; then
              git commit -m "[NEW DROP] war.gov CSV changed: ${{ steps.poll.outputs.row_count }} rows (${{ steps.poll.outputs.delta }} delta) via ${{ steps.poll.outputs.source }}"
            elif [ "${{ steps.poll.outputs.csv_stuck }}" = "true" ]; then
              git commit -m "poll: csv unreachable for ${{ steps.poll.outputs.consecutive_failures }} consecutive polls"
            else
              git commit -m "poll: war.gov unchanged ($(date -u +%FT%TZ))"
            fi
            git push
          fi

      - name: Open NEW DROP issue
        if: steps.poll.outputs.new_drop == 'true'
        uses: actions/github-script@v7
        with:
          script: |
            const rowCount = '${{ steps.poll.outputs.row_count }}';
            const delta = '${{ steps.poll.outputs.delta }}';
            const sha = '${{ steps.poll.outputs.sha256 }}';
            const source = '${{ steps.poll.outputs.source }}';
            const body = [
              '## war.gov uap-csv.csv has changed',
              '',
              `- New row count: **${rowCount}**`,
              `- Delta: **${delta}**`,
              `- New CSV SHA-256: \`${sha}\``,
              `- Fetched via: \`${source}\``,
              `- Detected: ${new Date().toISOString()}`,
              '',
              '### Next steps',
              '1. Pull latest: `git pull`',
              '2. The new CSV is at `_scratch/uap-csv.csv`',
              '3. Run pipeline: `python -m pipeline.run all`',
              '4. Review and push',
              '',
              'Triggered automatically by `.github/workflows/poll-wargov.yml`.',
            ].join('\n');
            await github.rest.issues.create({
              owner: context.repo.owner,
              repo: context.repo.repo,
              title: `[NEW DROP] war.gov UAP CSV changed (${delta} new rows, ${rowCount} total)`,
              body,
              labels: ['new-drop', 'automated'],
            });

      - name: Open CSV-unreachable issue (one-time)
        if: steps.poll.outputs.csv_stuck == 'true'
        uses: actions/github-script@v7
        with:
          script: |
            const failures = '${{ steps.poll.outputs.consecutive_failures }}';
            const body = [
              '## war.gov CSV unreachable',
              '',
              `The poller has failed to fetch \`uap-csv.csv\` on **${failures} consecutive polls** (~6 hours).`,
              '',
              'Both fetch paths returned blocked or empty responses:',
              '- `curl_cffi` with chrome120/119/116/edge101 impersonation',
              '- Headless Playwright Chromium with realistic Chrome fingerprint',
              '',
              'This is likely Akamai bot detection on war.gov\'s edge tightening',
              'against the GitHub Actions runner IP pool. Possible remediations:',
              '',
              '1. Add a residential proxy step (Bright Data, ScrapingBee)',
              '2. Run the poller from a different IP (Azure runner, DigitalOcean)',
              '3. Wait it out (Akamai reputation scores decay)',
              '',
              'No further alerts will be opened until CSV is reachable again.',
              'Once reachable, the failure counter resets.',
              '',
              'Triggered automatically by `.github/workflows/poll-wargov.yml`.',
            ].join('\n');
            await github.rest.issues.create({
              owner: context.repo.owner,
              repo: context.repo.repo,
              title: `[STUCK] war.gov CSV unreachable for ${failures} consecutive polls`,
              body,
              labels: ['csv-stuck', 'automated', 'investigate'],
            });

      - name: Fail if poll errored
        if: steps.poll.outcome == 'failure' && steps.poll.outputs.new_drop != 'true' && steps.poll.outputs.csv_stuck != 'true'
        run: |
          echo "::error::Poll script failed (network/parse error). Investigate logs above."
          exit 1
